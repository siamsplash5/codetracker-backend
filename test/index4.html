#include <bits/stdc++.h>
using namespace std;
using ll = long long;
using P = pair<int,int>;
#define rep(i,n) for(int i=0; i<n; i++)
const ll INF=(1ll<<60);
const int MOD = 1e9 + 7;
 
int main(){
    /*入力*/
    int N,K;
    cin >> N >> K;
    int P[N],C[N];
    rep(i,N) cin >> P[i];
    rep(i,N) cin >> C[i];
    ll ans = -1e18;

    /*ループ(周期性)を考える*/
    vector<vector<int>> route(N);

    /*スタートを1からNまですべてのマスで考える*/
    rep(i,N){
        int now = P[i];
        set<int> st;
        vector<int> score;
        st.insert(i+1);
        route[i].push_back(i+1);
        while(st.count(now) == 0){
            st.insert(now);
            route[i].push_back(now);
            score.push_back(C[now-1]);
            now = P[now-1];
        }
        route[i].push_back(now);
        score.push_back(C[now-1]);
        /*ループの長さやその間にゲットできる得点を計算する*/
        int loop;
        ll getscore=0;
        int first;
        rep(j,route[i].size()-1) if(route[i][j]==route[i][route[i].size()-1]){
            first = j;
            loop = route[i].size()-1-j;
            break;
        }
        for(int k=first; k<route[i].size(); k++) getscore+=score[k];
        /*ループの長さ以下なら普通にやるだけ、ループ以上ならうまくやる*/
        ll SCORE = 0;
        if(K < loop+first){
            int start = i;
            rep(j,K){
                start = P[start]-1;
                SCORE += C[start];
                ans = max(ans,SCORE);
            }
        }else{
            int start = i;
            rep(j,first){
                start = P[start]-1;
                SCORE += C[start];
                ans = max(ans,SCORE);
            }
            int nokori = K-first;
            if(getscore >= 0){
                SCORE += getscore*(K/loop);
                int amari = K%loop;
                ans = max(ans,SCORE);
                rep(j,amari){
                    cout << score[j];
                    SCORE += score[j];
                    ans = max(ans,SCORE);
                }
            }else{
                rep(j,loop){
                    SCORE += score[j];
                    ans = max(ans,SCORE);
                }
            }
        }
        ans = max(ans,SCORE);
    }
    
    /*出力*/
    cout << ans << endl;
 
    return 0;
}
