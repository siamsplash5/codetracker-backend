#include <bits/stdc++.h>
using namespace std;
typedef long long int lld;

int main() {
    lld n; cin >> n;
    lld a[n + 2];
    bool mark[n + 2] = {0};
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    lld c = 0, pos = 1;
    while (1) {
        if (pos == 2)break;
        if (mark[pos]) {
            c = -1;
            break;
        }
        mark[pos] = 1;
        pos = a[pos];
        c++;
    }
    cout << c << endl;
    return 0;
}
